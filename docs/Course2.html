ipak <- function(pkg, repository = c("CRAN", "Bioconductor", "github")) {
    new.pkg <- pkg[!(pkg %in% installed.packages()[, "Package"])]
    if (length(new.pkg)) {
        if (repository == "CRAN") {
            install.packages(new.pkg)
        }
        if (repository == "Bioconductor") {
          if (!requireNamespace("BiocManager", quietly = TRUE)) {
            install.packages("BiocManager")
          }
          BiocManager::install(new.pkg)
        }
    }
}

Another effect of normlization of rlog is to shrink the variance of low expression gene. We can see how genes with low counts (bottom left-hand corner) seem to be excessively variable on the ordinary logarithmic scale, while the rlog compress differences for the low count genes for which the data provide little information about differential expression.

dds <- estimateSizeFactors(dds)
df <- bind_rows(
  as_data_frame(log2(counts(dds, normalized=TRUE)[, 1:2]+1)) %>%
    mutate(transformation = "log2(x + 1)"),
  as_data_frame(assay(rld)[, 1:2]) %>%
    mutate(transformation = "rlog"))
colnames(df)[1:2] <- c("x", "y")  
ggplot(df, aes(x = x, y = y)) + geom_hex(bins = 80) +
  coord_fixed() + facet_grid( . ~ transformation) + theme_bw()

4.3 Clustering

It is useful to select high variable genes (HVG) to reduce the technical noise. HVG can be calculated by standard deviation of genes across samples. Here the top 500 HVGs are selected for clustering.

topVarGenes = head(order(rowVars(assay(rld)), decreasing = TRUE), 500)

Hierarchical clustering of samples.

correlationMatrix = cor(assay(rld)[topVarGenes, ])
sampleDistMatrix = 1 - correlationMatrix
rownames(correlationMatrix) = rownames(colData)
colnames(correlationMatrix) = rownames(colData)
pheatmap(correlationMatrix,
         clustering_distance_rows = as.dist(sampleDistMatrix),
         clustering_distance_cols = as.dist(sampleDistMatrix))

4.4 PCA

PCA is to find the most variable direction in the data, which called principle component (PC). In general the samples can be separated in the first few PCs, help us to visualize the data in low dimensions and understand the sample similarity.
Similarly, we use the top 500 HVG to perform PCA.

pca = prcomp(t(assay(rld)[topVarGenes, ]), center = TRUE, scale. = FALSE)
xy = as.data.frame(pca$x)
expVar = round((pca$sdev)^2/sum((pca$sdev)^2) * 100)
expVar = paste0(colnames(xy), ":", expVar, "% variance")
xy$Condition = colData$condition
ggplot(xy, aes(PC1, PC2)) + 
  geom_point(aes(colour=factor(Condition)), size = 3) + 
  theme_bw() +
  labs(color = "Condition", title = "OSK", x = expVar[1], y = expVar[2])

6 Plotting results

6.1 Heatmap of DEG

We first sign the marker genes to each condition accroding to the log2FoldChange using ifelse function. We then pick out the top 30 significant DEG for each condition.

res = as.data.frame(res[complete.cases(res), ])
res$genes = rownames(res)
res$condition = ifelse(res$log2FoldChange > 0, "ESC", "MEF")
markers = res %>% group_by(condition) %>% top_n(-30, padj)

Scale the gene across samples to mean = 0 and standard deviation = 1, namely z-score. Z-score often generates better visualization for gene expression heatmap.

norm.exp = assay(rld[markers$genes, colnames(sub_dds)])
scale.exp = t(scale(t(norm.exp)))
pheatmap(scale.exp,fontsize_row=6)

7.2 Gene Set Enrichment Analysis (GSEA)

Use the fold change to perform GSEA. By default, GSEA uses Reactome pathway database for enrichment - gsePathway. Althernately, one can use gseGO to perform GSEA using GO database.

deg = res
gid = bitr(deg$genes, fromType = "SYMBOL", toType = "ENTREZID", OrgDb = "org.Mm.eg.db")
deg$gene_id = as.numeric(gid$ENTREZID[match(deg$genes,gid$SYMBOL)])
deg = deg[complete.cases(deg),]
geneList = structure(deg$log2FoldChange, names = deg$gene_id)
geneList = sort(geneList, decreasing = TRUE)
gsea = gsePathway(geneList, nPerm=10000,organism = "mouse",
                pvalueCutoff=0.1,
                pAdjustMethod="BH", verbose=FALSE)

Traslate the gene id to gene symbol.

enrich_res = gsea@result
newGeneNames = sapply(enrich_res$core_enrichment, function(gs) {
  unlist(paste0(gid$SYMBOL[match(strsplit(gs, '/')[[1]], gid$ENTREZID)],collapse = '/'))
})
enrich_res$core_enrichment = newGeneNames
enrich_res = enrich_res[order(enrich_res$NES, decreasing = TRUE), ]

The gene sets with positive Normalized Enrichment Score (NES) are ESC enriched pathways, while gene sets with negative NES are MEF enriched pathways. Below I select two pathways for show.
